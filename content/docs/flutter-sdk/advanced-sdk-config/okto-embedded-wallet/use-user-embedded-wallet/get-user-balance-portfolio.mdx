          Future<UserPortfolioResponse>? _userPortfolio;
        
          Future<UserPortfolioResponse> getuserPortfolio() async {
            try {
              final userPortfolio = await okto!.userPortfolio();
              return userPortfolio;
            } catch (e) {
              throw Exception(e);
            }
          }
